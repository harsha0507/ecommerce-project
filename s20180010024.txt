	Address(String street,String city,int pin)
	{
		this.street = street;
		this.city = city;
		this.pin = pin;
	}
	Address(String street,String city)
	{
		this.street = street;
		this.city = city;
	}
	void set(String street,String city,int pin)
	{
		this.street = street;
		this.city = city;

		this.street = street;

}
